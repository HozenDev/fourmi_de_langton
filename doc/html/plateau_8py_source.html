100 """!@brief valide start & end in schema
101 @param start list of int index where start to draw (default (0, 0))
102 @param end list of int index where end to draw (default (None, None))
103 @return a valid couple of start & end (valid index)
104 """
105 if end[0] == None: # if end_x not setting in draw()
106 end[0] = self.w
107 if end[1] == None: # if end_y not setting in draw()
108 end[1] = self.h
109 else :
110 if start[0] < 0:
111 start[0] = start[0]%self.w
112 if start[1] < 0:
113 start[1] = start[1]%self.h
114 if end[0] > self.w:
115 end[0] = end[0]%self.w
116 if end[1] > self.h:
117 end[1] = end[1]%self.h
118 return start, end
