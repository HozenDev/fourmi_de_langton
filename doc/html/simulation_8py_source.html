150 def add_fourmi(self):
151 """!@brief Add an ant on the grid"""
152 if not self.run :

156 for event in pygame.event.get():

158 not_click = False

160 and event.pos[1] >= 0 and event.pos[1] < self.size_plateau[1] :
161 e_x, e_y = event.pos
162 f_x = (e_x//self.res)*self.res
163 f_y = (e_y//self.res)*self.res
164 f_new = lgt.Fourmi(coords=(f_x, f_y),
165 taille=self.res,
166 speed=self.res,
167 direction=0,
168 behavior=self.behavior,
169 color=self.color)
170 self.fourmi_list.append(f_new)
171 self.nb_fourmi += 1
172 f_new.draw()
173 else:
174 print("Can't create an ant here.")

199 def next_step(self):
200 """!@brief Play simulation iterator next_number times, (default 1)
201 @exception StopIteration If Simulation ended
202 @exception Exception If Simulation already run
203 """
204 try:
205 if not self.end and not self.run:
206 for _ in range(self.next_time):
207 next(self.it)

209 print("Simulation ended")
